Strukturvorschlag: folgender grober Programmablauf (analgo zu TS-A, -B):

program.cs ruft Zentrales Steuerelement: MeshDataGenerationRunner auf
- MeshDataGenerationRunner ruft ein "Zeitpunktermittlungsmodul" auf: bisher sind das Event Runner. Jetzt ist es lediglich ein Erzeugungsmodul (Factory?), das für die Epochen 1, 2, 3 die Zeitpunkte als JD in UTC erzeugt
- danach kommt ein BatchingGenerator. Dieser muss Daten für effiziente API Aufrufe erstellen lassen. Jeder Planet wird nacheinander für die einzelnen Epcohen abgearbeitet. Die Anzahl der Aufrufe an die Horizons API muss limitiert werden. Der Generator liefert "Planet | Startwert, Endwert, Schrittweite ", so dass insgesamt eine überschaubare Anzahl an Aufrufen resultiert. Für jeden Chunk ruft der BatchingGeneator eine "MeshAPI-RequestFactory" auf 
- Die "MeshAPI-RequestFactory" verwendet den Eingang "Planet | Startwert, Endwert, Schrittweite" und erweitert die Datenstruktur um die Parameter für die Simulation: helioecliptic.
- Weiterreichung des Requests an den HorizonsAPIClient, der die Anfragen stellt, wait state 500ms einbauen
- Wir bekommen ein Horizons Ergebnis zurück, eine strukturierte Liste von Zeitpunkten und (X, Y, Z, VX, VY, VZ): jetzt verwenden wir dieselbe Kette wie TS-A und TS-B mit MeshDataL0RawExportRunner (CSV), MeshDataL0xportRunner (Json).

using EphemerisRegression.Api;
using EphemerisRegression.Config;
using EphemerisRegression.Domain;
using EphemerisRegression.Export;
using EphemerisRegression.Infrastructure;
using EphemerisRegression.Parsing;
using EphemerisRegression.Util;
using EphemerisRegression.Domain;
using EphemerisRegression.Api;
using EphemerisRegression.Config;
using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;
using System.Threading.Tasks;

namespace EphemerisRegression.Runner
{
    public sealed class MeshDataGenerationRunner
    {
        private const int MaxStepsPerChunk = 2000;

        private readonly HorizonsApiClient _apiClient = new();


        private readonly MeshHorizonsApiRequestFactory _meshFactory;

        public MeshDataGenerationRunner()
        {
            var config = new HorizonsHelioLevel0Config();
            _meshFactory = new MeshHorizonsApiRequestFactory(config);
        }

        public async Task RunEarthAsync()
        {
            Console.WriteLine("TS-D Earth automatic run started...");
            Console.WriteLine();

            var solutionRoot = ProjectPathResolver.GetSolutionRoot();

            string rawDir = Path.Combine(
                solutionRoot,
                "EphemerisRegression",
                "Horizons",
                "Mesh",
                "Helio",
                "L0",
                "Raw");

            string jsonDir = Path.Combine(
                solutionRoot,
                "EphemerisRegression",
                "Horizons",
                "Mesh",
                "Helio",
                "L0",
                "Json");

            Directory.CreateDirectory(rawDir);
            Directory.CreateDirectory(jsonDir);

            var config = new HorizonsHelioLevel0Config();
            var meshFactory = new MeshHorizonsApiRequestFactory(config);

            var epochs = new[]
            {
                ("Core",     1600, 1, 1, 2400, 1, 1,  30.0),
                ("Extended",    0, 1, 1, 4000, 1, 1, 180.0),
                ("Extreme",  -4000, 1, 1, 8000, 1, 1, 730.0)
            };

            foreach (var e in epochs)
            {
                Console.WriteLine($"Epoch: {e.Item1}");

                double jdStart = ToJulianDayProlepticGregorian(e.Item2, e.Item3, e.Item4);
                double jdStop = ToJulianDayProlepticGregorian(e.Item5, e.Item6, e.Item7);

                double stepDays = e.Item8;

                var allVectors = new List<StateVector>();
                var allRequests = new List<(string Canonical, string Hash)>();

                int chunkIndex = 1;

                for (double chunkStartJD = jdStart;
                     chunkStartJD <= jdStop;
                     chunkStartJD += MaxStepsPerChunk * stepDays, chunkIndex++)
                {
                    double chunkStopJD = chunkStartJD + (MaxStepsPerChunk - 1) * stepDays;
                    if (chunkStopJD > jdStop)
                        chunkStopJD = jdStop;

                    var request = meshFactory.Create(
                        399,
                        chunkStartJD,
                        chunkStopJD,
                        stepDays);

                    Console.WriteLine($"  Chunk {chunkIndex:00}: JD{chunkStartJD} -> JD{chunkStopJD}");

                    string rawContent = await _apiClient.ExecuteAsync(request);

                    string rawFileName = $"Earth_TS-D_L0_{e.Item1}_Chunk_{chunkIndex:00}.csv";
                    string rawPath = Path.Combine(rawDir, rawFileName);

                    await File.WriteAllTextAsync(rawPath, rawContent);

                    var parser = new HorizonsVectorParser();
                    var vectors = parser.Parse(rawContent).ToList();

                    if (vectors.Count == 0)
                    {
                        throw new InvalidOperationException(
                            $"Horizons returned 0 vectors for {e.Item1} chunk {chunkIndex:00}. " +
                            $"Check RAW file: {rawPath}");
                    }

                    allVectors.AddRange(vectors);

                    var parameters = request.ToParameterDictionary();
                    var canonical = CanonicalRequestBuilder.Build(parameters);
                    var hash = HashCalculator.ComputeSha256(canonical);

                    allRequests.Add((canonical, hash));
                }

                string epochHash = HashCalculator.ComputeSha256(
                    string.Join("|", allRequests.Select(r => r.Hash)));

                var metadata = new ReferenceMetadata
                {
                    EpochHash = epochHash,
                    Requests = allRequests.Select(r => new RequestInfo
                    {
                        CanonicalRequest = r.Canonical,
                        RequestHash = r.Hash,
                        HorizonsUrl = ""
                    }).ToList(),
                    HorizonsUrl = "",
                    EngineVersion = "1.1.0",
                    CorrectionLevel = "L0",
                    Mode = "HELIO",
                    EphemType = "VECTORS",
                    GeneratedAtUtc = DateTime.UtcNow
                };

                var reference = new ReferenceStateVector(
                    "Earth",
                    "TS-D",
                    e.Item1,
                    "L0",

                Console.WriteLine();
            }

        private static double ToJulianDayProlepticGregorian(int year, int month, int day)
        {
            int a = (14 - month) / 12;
            int y = year + 4800 - a;
            int m = month + 12 * a - 3;

            int jdn = day
                + (153 * m + 2) / 5
                + 365 * y
                + y / 4
                - y / 100
                + y / 400
                - 32045;

            return jdn - 0.5;
        }

        private async Task RunPlanetEpochAsync(
    PlanetInfo planet,
    string epochName,
    int startY, int startM, int startD,
    int stopY, int stopM, int stopD,
    double stepDays,
    string rawDir,
    string jsonDir,
    bool overwrite)
        {
            Console.WriteLine($"Planet: {planet.Name} | Epoch: {epochName}");

            double jdStart = ToJulianDayProlepticGregorian(startY, startM, startD);
            double jdStop = ToJulianDayProlepticGregorian(stopY, stopM, stopD);

            var allVectors = new List<StateVector>();
            var allRequests = new List<(string Canonical, string Hash)>();

            int chunkIndex = 1;

            for (double chunkStartJD = jdStart;
                 chunkStartJD <= jdStop;
                 chunkStartJD += MaxStepsPerChunk * stepDays, chunkIndex++)
            {
                double chunkStopJD = chunkStartJD + (MaxStepsPerChunk - 1) * stepDays;
                if (chunkStopJD > jdStop) chunkStopJD = jdStop;

                var request = _meshFactory.Create(
                    planet.CommandCode,
                    chunkStartJD,
                    chunkStopJD,
                    stepDays);

                string rawFileName = $"{planet.Name}_TS-D_L0_{epochName}_Chunk_{chunkIndex:00}.csv";
                string rawPath = Path.Combine(rawDir, rawFileName);

                Console.WriteLine($"  Chunk {chunkIndex:00}: {request.StartTime} -> {request.StopTime} | STEP={request.StepSize}");

                string rawContent = await _apiClient.ExecuteAsync(request);
                await File.WriteAllTextAsync(rawPath, rawContent);

                var parser = new HorizonsVectorParser();
                var vectors = parser.Parse(rawContent).ToList();

                if (vectors.Count == 0)
                    throw new InvalidOperationException($"0 vectors: {planet.Name} {epochName} chunk {chunkIndex:00}. RAW={rawPath}");

                allVectors.AddRange(vectors);

                var parameters = request.ToParameterDictionary();
                var canonical = CanonicalRequestBuilder.Build(parameters);
                var hash = HashCalculator.ComputeSha256(canonical);

                allRequests.Add((canonical, hash));
            }

            string epochHash = HashCalculator.ComputeSha256(
                string.Join("|", allRequests.Select(r => r.Hash)));

            var metadata = new ReferenceMetadata
            {
                EpochHash = epochHash,
                Requests = allRequests.Select(r => new RequestInfo
                {
                    CanonicalRequest = r.Canonical,
                    RequestHash = r.Hash,
                    HorizonsUrl = ""
                }).ToList(),
                HorizonsUrl = "",
                EngineVersion = "1.1.0",
                CorrectionLevel = "L0",
                Mode = "HELIO",
                EphemType = "VECTORS",
                GeneratedAtUtc = DateTime.UtcNow
            };

            var reference = new ReferenceStateVector(
                planet.Name,
                "TS-D",
                epochName,
                "L0",
                allVectors,
                metadata);

            var writer = new JsonReferenceWriter(JsonOptionsFactory.Create());

            string jsonPath = Path.Combine(jsonDir, $"{planet.Name}_TS-D_L0_{epochName}.json");
            await writer.WriteAsync(jsonPath, reference);

            Console.WriteLine($"  JSON saved: {Path.GetFileName(jsonPath)} | vectors={allVectors.Count}");
            Console.WriteLine();
        }

using System;
using System.Collections.Generic;
using System.Linq;

namespace EphemerisRegression.Domain
{
    public sealed record PlanetInfo(string Name, int CommandCode)
    {
        public override string ToString() => $"{Name} ({CommandCode})";
    }

    public static class PlanetCatalog
    {
        public static readonly IReadOnlyList<PlanetInfo> AllPlanets =
            new List<PlanetInfo>
            {
                new("Mercury", 199),
                new("Venus",   299),
                new("Earth",   399), // IMPORTANT: Earth, NOT EMB
                new("Mars",    499),
                new("Jupiter", 599),
                new("Saturn",  699),
                new("Uranus",  799),
                new("Neptune", 899)
            };

        public static PlanetInfo Earth =>
            AllPlanets.Single(p => p.Name == "Earth");

        public static IReadOnlyList<PlanetInfo> ParseSelection(string? selection)
        {
            if (string.IsNullOrWhiteSpace(selection))
                return AllPlanets;

            var wanted = selection
                .Split(new[] { ',', ';' }, StringSplitOptions.RemoveEmptyEntries)
                .Select(s => s.Trim())
                .ToHashSet(StringComparer.OrdinalIgnoreCase);

            var result = AllPlanets
                .Where(p => wanted.Contains(p.Name))
                .ToList();

            if (result.Count != wanted.Count)
                throw new ArgumentException(
                    "Unknown planet(s). Known: " +
                    string.Join(", ", AllPlanets.Select(p => p.Name)));

            return result;
        }
    }
}

using System.Collections.Generic;

namespace EphemerisRegression.Api
{
    public sealed class HorizonsApiRequest
    {
        public int Command { get; init; }
        public string StartTime { get; init; } = "";
        public string StopTime { get; init; } = "";
        public string StepSize { get; init; } = "1h";

        public string Center { get; init; } = "@10";
        public string RefPlane { get; init; } = "ECLIPTIC";
        public string RefSystem { get; init; } = "ICRF";
        public string? VectorCorrection { get; init; }
        public string OutputUnits { get; init; } = "AU-D";

        public string CsvFormat { get; init; } = "NO";

        public IDictionary<string, string> ToParameterDictionary()
        {
            var dict = new Dictionary<string, string>
            {
                ["COMMAND"] = Command.ToString(),
                ["CENTER"] = Center,
                ["START_TIME"] = StartTime,
                ["STOP_TIME"] = StopTime,
                ["STEP_SIZE"] = StepSize,
                ["REF_PLANE"] = RefPlane,
                ["REF_SYSTEM"] = RefSystem,
                ["OUT_UNITS"] = OutputUnits,
                ["EPHEM_TYPE"] = "VECTORS",
                ["CSV_FORMAT"] = CsvFormat,
                ["OBJ_DATA"] = "NO"
            };

            if (!string.IsNullOrWhiteSpace(VectorCorrection))
                dict["VECT_CORR"] = VectorCorrection!;

            return dict;
        }
    }
}

using EphemerisRegression.Batching;

    public sealed class MeshHorizonsRequestFactory
    {
        private readonly HorizonsLevelBaseConfig _config;

        public MeshHorizonsRequestFactory(HorizonsLevelBaseConfig config)
        {
            _config = config;
        }

        public (HorizonsApiRequest Request, string Canonical, string Hash)
            Create(MeshBatchDefinition batch)
        {

            {
                Command = batch.PlanetCode,
                StartTime = batch.StartUtc.ToIsoString(),
                StopTime = batch.StopUtc.ToIsoString(),
                StepSize = $"{batch.StepDays} d",

                Center = _config.Center,
                RefPlane = _config.RefPlane,
                RefSystem = _config.RefSystem,
                VectorCorrection = _config.VectorCorrection,
                OutputUnits = _config.OutputUnits
            };

            var parameters = request.ToParameterDictionary();

            var canonical = CanonicalRequestBuilder.Build(parameters);
            var hash = HashCalculator.ComputeSha256(canonical);

            return (request, canonical, hash);
        }
    }
