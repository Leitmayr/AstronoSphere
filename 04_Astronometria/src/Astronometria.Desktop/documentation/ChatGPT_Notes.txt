In der Astronometria gibt es zwei Zeit-Domänen: 
1) die UserTime mit UTC, Zeitzonen und Sommerzeit.
2) die AstroTime mit TT (später vlt. mit TDC)

Im Moment sind diese Zeitdomänen in einer Klasse AstroTime vermischt. Es soll vermieden werden, dass in der Astro-Domain die UTC verwendet wird und in der User Domain die TT. 
In der angehängten Datei "DieZeitInDerAstronometria.ppt" ist die grobe Architektur des Programms dargestellt und die beiden Fundamentaldomänen User und Astro. Schau Dir diesen Foliensatz sehr intensiv an!



Ich habe mir die folgende Architektur überlegt:
Die PPTX enthält zwei Interface-Klassen in den beiden Domänen. Domänenübergreifend soll nur über das Interface kommuniziert werden. Von User nicht direkt mit der AstroIF Klasse, von Astro nicht mit der UserIF Klasse. Innerhalb ihrer Domäne haben die beiden Interface Klassen als einzige Zugriff auf die "ihre" Zeit. Die einzelnen Klassen innerhalb der Domänen müssen auch durch "ihre" IF-Klassen.

Diese Architektur hat Lücken/Schwächen: wird z.B. in einem Report auf AstroTime Basis eine Zeitreihe mit Koordinaten erzeugt, so muss vor Veröffentlichung als XML eine Umrechnungsmethode ausgeführt werden, ohne dass die bestehende Zeitreihe mit ihren Daten korrumpiert wird. Vermutlich ist etwas wie eine Factory nötig, um die Zeitreihe für die User Domäne zu erzeugen.

		using System;

namespace AstroSim.Core.Time
{
    /// <summary>
    /// Represents an astronomical time instant.
    /// Internally supports UT (JD_UT) and TT (JD_TT).
    /// Designed for deterministic ephemeris calculations (VSOP).
    /// </summary>
    public sealed class AstroTime
    {
        private const double JD_J2000 = 2451545.0;
        private const double SecondsPerDay = 86400.0;
        private const double DaysPerJulianCentury = 36525.0;
        private const double DaysPerJulianMillennium = 365250.0;

        // --- Public Properties ---
        public double JD_UT { get; }
        public double JD_TT { get; }

        // ============================================================
        // === CONSTRUCTORS ===========================================
        // ============================================================

        /// <summary>
        /// Constructs AstroTime from a UTC/UT calendar date.
        /// </summary>
        public AstroTime(int year, int month, int day,
                         int hour, int minute, int second)
        {
            double hourDecimal = hour + minute / 60.0 + second / 3600.0;

            JD_UT = GregorianToJulianDate(year, month, day, hourDecimal);

            double deltaT = DeltaT(JD_UT, year);

            JD_TT = JD_UT + deltaT / SecondsPerDay;
        }

        /// <summary>
        /// Constructs AstroTime directly from Julian Date in TT.
        /// Useful for VSOP test data (vsop87.chk).
        /// </summary>
        public AstroTime(double jdTT)
        {
            JD_TT = jdTT;

            // Optional UT reconstruction (approximate)
            double approxYear = JulianDateToYear(jdTT);
            double deltaT = DeltaT(jdTT, (int)Math.Round(approxYear));

            JD_UT = JD_TT - deltaT / SecondsPerDay;
        }

        // ============================================================
        // === PUBLIC VSOP HELPERS ===================================
        // ============================================================

        /// <summary>
        /// Julian centuries since J2000.0 (TT).
        /// </summary>
        public double JulianCenturiesSinceJ2000()
        {
            return (JD_TT - JD_J2000) / DaysPerJulianCentury;
        }

        /// <summary>
        /// Julian millennia since J2000.0 (TT).
        /// Required for VSOP87.
        /// </summary>
        public double JulianMillenniaSinceJ2000()
        {
            return (JD_TT - JD_J2000) / DaysPerJulianMillennium;
        }

        // ============================================================
        // === INTERNAL TIME CONVERSION ================================
        // ============================================================

        private static double GregorianToJulianDate(int year, int month, int day, double hour)
        {
            if (month <= 2)
            {
                year -= 1;
                month += 12;
            }

            int A = (int)Math.Floor(year / 100.0);
            int B = 2 - A + (int)Math.Floor(A / 4.0);

            double jd =
                Math.Floor(365.25 * (year + 4716)) +
                Math.Floor(30.6001 * (month + 1)) +
                day + hour / 24.0 + B - 1524.5;

            return jd;
        }

        /// <summary>
        /// Rough year estimation from Julian Date.
        /// Only used internally for DeltaT approximation.
        /// </summary>
        private static double JulianDateToYear(double jd)
        {
            return 2000.0 + (jd - JD_J2000) / 365.25;
        }

        // ============================================================
        // === DELTA T =================================================
        // ============================================================

        /// <summary>
        /// Computes DeltaT (TT - UT) in seconds.
        /// Polynomial approximation sufficient for VSOP use.
        /// </summary>
        /// /// DeltaT (TT - UT) approximation.
        /// Sources:
        /// - Meeus, Astronomical Algorithms, 2nd ed., Chapter 10
        /// - NASA polynomial expressions (Espenak & Meeus)
        private static double DeltaT(double jd, int year)
        {
            double t = (jd - JD_J2000) / DaysPerJulianCentury;
            double y = 2000.0 + t * 100.0;

            if (year < 948)
                return 2177 + 497 * t + 44.1 * t * t;

            if (year < 1600)
                return 102 + 102 * t + 25.3 * t * t;

            if (year <= 2005)
            {
                double u = y - 2000.0;
                return 63.86 + u * (
                           0.3345 + u * (
                          -0.060374 + u * (
                           0.0017275 + u * (
                           0.000651814 + u * 0.00002373599))));
            }

            if (year <= 2050)
            {
                double u = y - 2000.0;
                return 62.92 + 0.32217 * u + 0.005589 * u * u;
            }

            double v = (year - 1820) / 100.0;
            return -20 + 32 * v * v;
        }
    }
}
